// 安裝vue-cli套件
npm install -g @vue/cli

// 解除安裝
npm uninstall -g @vue/cli

// 建立vue專案 命令提示字元
vue create holiday_dessert

// 移動到專案下
cd holiday_dessert

// 啟動專案
npm run serve

// 測試網址
http://localhost:8080/

// 發布專案 (建立在c:/xampp/htdcos裡面)
npm run build
